// 导入头像 import avatarMountain from '@/assets/avatars/avatar-mountain.svg' import avatarSun from '@/assets/avatars/avatar-sun.svg' // ... // 头像列表 const defaultAvatars = [ 'avatar-mountain', 'avatar-sun', 'avatar-leaf', 'avatar-star', 'avatar-cat', 'avatar-fox', 'avatar-owl', 'avatar-cloud', 'avatar-wave', 'avatar-sprout', 'avatar-moon', 'avatar-gem' ] // 动态引用 const getAvatarUrl = (name) => new URL(\`../assets/avatars/\${name}.svg\`, import.meta.url).href
